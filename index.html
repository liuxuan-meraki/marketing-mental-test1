<script>

    // 增加选项点击效果：为选中的 Label 添加样式

    document.querySelectorAll('.option-label').forEach(label => {

        label.addEventListener('click', function() {

            const name = this.querySelector('input').name;

            // 清除同组已选样式

            document.querySelectorAll(`input[name="${name}"]`).forEach(input => {

                input.parentElement.classList.remove('selected');

            });

            // 添加当前选中样式

            this.classList.add('selected');

        });

    });

 

    function calculateResult() {

        const questions = ['q1', 'q2', 'q3', 'q4', 'q5'];

        let totalScore = 0;

        let allAnswered = true;

 

        // 1. 检查是否全部选择并累计分数

        questions.forEach(q => {

            const selected = document.querySelector(`input[name="${q}"]:checked`);

            if (!selected) {

                allAnswered = false;

            } else {

                totalScore += parseInt(selected.value);

            }

        });

 

        // 2. 校验逻辑

        if (!allAnswered) {

            alert("请完成所有题目后提交，你的心理健康很重要！");

            return;

        }

 

        // 3. 结果判断逻辑

        const resultArea = document.getElementById('result-area');

        const scoreDisplay = document.getElementById('score-display');

        const titleDisplay = document.getElementById('title-display');

        const adviceDisplay = document.getElementById('advice-display');

 

        scoreDisplay.innerText = `得分：${totalScore}`;

 

        if (totalScore <= 8) {

            titleDisplay.innerText = "�� 状态良好";

            adviceDisplay.innerHTML = "你的职场心态非常稳健！你能够很好地平衡工作压力与个人生活。建议继续保持当下的节奏，定期进行复盘，你还有很大的潜能可以挖掘。";

        } else if (totalScore <= 15) {

            titleDisplay.innerText = "�� 轻度倦怠";

            adviceDisplay.innerHTML = "你正承受着不小的营销压力。虽然目前还能应对，但已经出现了情绪波动和疲惫。建议：<br>1. 尝试与同事/朋友倾诉压力；<br>2. 强迫自己下班后断联1小时；<br>3. 寻找一个小爱好来转移注意力。";

        } else {

            titleDisplay.innerText = "�� 重度倦怠";

            adviceDisplay.innerHTML = "警告：你可能正处于严重的心理内耗中。高强度的KPI和内卷已经严重影响了你的心理健康。建议：<br>1. 立即申请短假或彻底断网休息；<br>2. 与主管沟通调整KPI或资源支持；<br>3. 考虑寻求专业心理咨询，记住：工作只是生活的一部分。";

        }

 

        // 4. 显示结果并平滑滚动

        resultArea.style.display = 'block';

        resultArea.scrollIntoView({ behavior: 'smooth' });

        

        // 隐藏提交区域防止重复操作

        document.getElementById('quiz-form').style.opacity = '0.5';

        document.getElementById('quiz-form').style.pointerEvents = 'none';

    }
